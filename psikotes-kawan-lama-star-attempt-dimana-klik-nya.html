perjalanan kota A ke kota B

#bus

stasiun A - stasiun X - stasiun B

jarak antar stasiun = 40 menit

total jarak antar stasiun

= 40 menit + 40 menit

= 80 menit

waktu nunggu bus = 20 menit

total waktu nunggu

= 20 menit + 20 menit

= 40 menit

total waktu dari stasiun A ke stasiun B

= 80 + 40

= 120 menit

#

#kereta

stasiun A - x stasiun - stasiun B

jarak antar stasiun = 30 menit

total jarak antar stasiun

= 30x menit

berhenti tiap stasiun = 15 menit

total berhenti

= 15x menit

waktu nunggu kereta = 10 menit

total waktu nunggu

= 10x menit

total waktu dari stasiun A ke stasiun B

= 30x + 15x + 10x

= 55x menit

#

waktu kereta = waktu bus

55x = 110

x = 110/55

x = 2
